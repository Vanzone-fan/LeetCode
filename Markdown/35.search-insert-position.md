// @lc code=start
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var searchInsert = function(nums, target) {
    // nums includes target
    // return index of target
    // nums does not include target
    // return index of smaller element
    if(nums.includes(target)){
        return nums.indexOf(target);
    }else{
        nums.push(target);
        nums.sort((a,b) => a-b);
        return nums.indexOf(target);
    }
    // 时间复杂度是O(nlogn) 不满足要求
};
// @lc code=end
